CREATE OR REPLACE TRIGGER MANOJ.T_TRIGGER
AFTER INSERT ON Staff
FOR EACH ROW
DECLARE A int;
BEGIN
    IF (:OLD.Bus_No=:NEW.Bus_No) THEN SELECT SID INTO A FROM Staff;
    DELETE FROM Staff WHERE SID=A;
    END IF;
END;


CREATE OR REPLACE TRIGGER MANOJ.T_TRIGGER1
AFTER INSERT
ON Staff
FOR EACH ROW
BEGIN
INSERT INTO DEPOT VALUES(:NEW.Bus_No,8,'Yeshwanthpur',:NEW.SID);
END;
