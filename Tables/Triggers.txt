Trigger 1
CREATE OR REPLACE TRIGGER MANOJ.T_TRIGGER1
AFTER INSERT
ON Staff
FOR EACH ROW
BEGIN
INSERT INTO DEPOT VALUES(:NEW.Bus_No,8,'Yeshwanthpur',:NEW.SID);
END;

Trigger 2
CREATE OR REPLACE TRIGGER MANOJ.T_TRIGGER
AFTER DELETE ON Staff
FOR EACH ROW
BEGIN
    DELETE FROM Depot WHERE SID=:OLD.SID;
END;

Stored Procedure
CREATE OR REPLACE PROCEDURE MANOJ.adjust_salary
IS
BEGIN
    UPDATE Staff
    SET Staff.Salary = Staff.Salary+Staff.Salary*(10/100)
    WHERE Staff.Years_Exp>=20;
END;
